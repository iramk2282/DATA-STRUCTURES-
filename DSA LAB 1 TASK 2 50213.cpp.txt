#include <iostream>
using namespace std;

int main() {
    int arr[] = {1, 2, 3, 4};
    int size = 4;

    // Reversing the array
    for (int i = 0; i < size / 2; i++) {
        int temp = arr[i];
        arr[i] = arr[size - 1 - i];
        arr[size - 1 - i] = temp;
    }

    // Display reversed array
    cout << "Reversed Array: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
